```csharp
namespace DiskScout.Models;

public enum PathCategory
{
    OsCriticalDoNotPropose,   // [1] HardBlacklist  -> forces Risk=Critique, Action=NePasToucher
    PackageCache,             // VC++ Redist caches, MSI Package Cache, Office C2R cache
    DriverData,               // %ProgramData%\Intel, NVIDIA, AMD driver data
    CorporateAgent,           // NinjaRMM, Bitdefender, Zscaler, Matrix42, Empirum, Splashtop
    VendorShared,             // %ProgramData%\Adobe, Autodesk, JetBrains shared workspaces
    Generic,                  // matched but no special semantic; lets KnownPathRule decorate score
}

public enum RiskLevel
{
    Aucun = 0,    // safe to delete
    Faible = 1,
    Moyen = 2,    // VerifierAvant
    Eleve = 3,
    Critique = 4, // NePasToucher
}

public enum RecommendedAction
{
    Supprimer,        // safe — matches RiskLevel.Aucun (score >= 80)
    CorbeilleOk,      // matches RiskLevel.Faible
    VerifierAvant,    // matches RiskLevel.Moyen
    NePasToucher,     // matches RiskLevel.Eleve / Critique
}

public sealed record PathRule(
    string Id,
    string PathPattern,           // glob-ish pattern, %ENVVAR% expansion + literal {Publisher} (deferred)
    PathCategory Category,
    RiskLevel? MinRiskFloor,      // null = no floor; PackageCache+CorporateAgent => Eleve
    string? Reason);              // human-readable, e.g. "Cache MSI Visual C++ Redistributable"

public sealed record RuleHit(string RuleId, PathCategory Category, string Reason);

public sealed record MatcherHit(string Source, string Evidence, int ScoreDelta);
// Source: "Registry"|"Service"|"Driver"|"Appx"|"ScheduledTask"|"Process"
// Evidence: e.g. "HKLM\SOFTWARE\Microsoft\NET" or "Service:wuauserv (Running)"
// ScoreDelta: negative number subtracted from the base 100 score (e.g. -50 for registry hit)
```
