<objective>
Build the **Residue Scanner**: post-uninstall scanner that finds artifacts left behind by a uninstalled program across registry, filesystem, shortcuts, MSI patches, services, scheduled tasks, and shell extensions. Inputs: target program identity (Publisher + DisplayName + InstallLocation) and OPTIONAL InstallTrace from Plan 01. Output: a list of `ResidueFinding`s pre-filtered through a strict system-path whitelist.

Purpose: This is the deep-scan engine. Together with publisher rules (Plan 04) and the native uninstaller (Plan 02), it gives the wizard everything it needs to propose deletions safely.
Output: `IResidueScanner` + `ResidueScanner` + `ResiduePathSafety` whitelist + tests.
</objective>

        public static readonly string[] CriticalFilesystemSubstrings =
        {

            @"\Windows\SysWOW64",
            @"\Windows\WinSxS",

            @"\Program Files (x86)\Windows NT",
            @"\Program Files\Microsoft\Edge",
            @"\Program Files (x86)\Microsoft\Edge",
            @"\Program Files\WindowsApps",
        };

        // Registry critical-key denylist (prefix, case-insensitive).
        public static readonly string[] CriticalRegistryPrefixes =
        {
            @"HKLM\SYSTEM\CurrentControlSet\Services\Tcpip",
            @"HKLM\SYSTEM\CurrentControlSet\Services\Dhcp",
            @"HKLM\SYSTEM\CurrentControlSet\Services\Dnscache",
            @"HKLM\SYSTEM\CurrentControlSet\Services\WinDefend",
            @"HKLM\SYSTEM\CurrentControlSet\Services\MsSecCore",
            @"HKLM\SYSTEM\CurrentControlSet\Services\WdNisSvc",
            @"HKLM\SYSTEM\CurrentControlSet\Services\Sense",
            @"HKLM\SOFTWARE\Microsoft\Windows Defender",
            @"HKLM\SOFTWARE\Microsoft\Windows Security",
            @"HKLM\SOFTWARE\Policies\Microsoft\Windows Defender",
            @"HKLM\SYSTEM\CurrentControlSet\Control\Class",
            @"HKLM\SYSTEM\Setup",
        };

        // AV/security service-name patterns to never touch.
        public static readonly string[] CriticalServiceNamePatterns =
        {
            "WinDefend", "MsSecCore", "WdNisSvc", "Sense", "WdFilter",
            "MpsSvc", "BFE", "EventLog", "RpcSs", "DcomLaunch",
            "TrustedInstaller",
        };

        public static bool IsSafeToPropose(string path)
        {
            if (string.IsNullOrWhiteSpace(path)) return false;
            foreach (var s in CriticalFilesystemSubstrings)
                if (path.Contains(s, StringComparison.OrdinalIgnoreCase)) return false;
            foreach (var p in CriticalRegistryPrefixes)
                if (path.StartsWith(p, StringComparison.OrdinalIgnoreCase)) return false;
            return true;
        }

        public static bool IsSafeServiceName(string serviceName)
        {
            if (string.IsNullOrWhiteSpace(serviceName)) return false;
            foreach (var pat in CriticalServiceNamePatterns)
                if (string.Equals(serviceName, pat, StringComparison.OrdinalIgnoreCase)) return false;
            return true;
        }
    }
