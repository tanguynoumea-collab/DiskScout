    ```csharp
    using System.Collections.ObjectModel;
    using CommunityToolkit.Mvvm.ComponentModel;
    using DiskScout.Models;

    namespace DiskScout.ViewModels.UninstallWizard;

    public sealed partial class ResidueTreeNode : ObservableObject
    {
        public string Label { get; init; } = "";
        public string? Path { get; init; }                         // null for category/group nodes
        public long SizeBytes { get; init; }
        public ResidueCategory Category { get; init; }
        public ResidueTrustLevel Trust { get; init; }
        public ResidueSource Source { get; init; }
        public string? Reason { get; init; }
        public ObservableCollection<ResidueTreeNode> Children { get; } = new();

        [ObservableProperty]
        private bool? _isChecked = false;        // tri-state for parents (null = mixed)

        public string SizeDisplay => DiskScout.Helpers.ByteFormat.Fmt(SizeBytes);
        public string TrustDisplay => Trust switch
        {
            ResidueTrustLevel.HighConfidence => "Confiance haute",
            ResidueTrustLevel.MediumConfidence => "Confiance moyenne",
            ResidueTrustLevel.LowConfidence => "Confiance basse",
            _ => "",
        };

        partial void OnIsCheckedChanged(bool? value)
        {
            if (Children.Count == 0 || value is null) return;
            foreach (var child in Children) child.IsChecked = value;
        }

        public IEnumerable<ResidueTreeNode> EnumerateLeavesChecked()
        {
            if (Path is not null && IsChecked == true) yield return this;
            foreach (var c in Children)
                foreach (var leaf in c.EnumerateLeavesChecked()) yield return leaf;
        }
    }
    ```

    ```csharp
    [RelayCommand]
    private async Task RunAsync()
    {
        IsRunning = true;
        OutputLines.Clear();
        try
        {
            var preferSilent = _wizard.PreferSilent;
            var quiet = ReadQuietUninstallString(_wizard.Target);
            var cmd = _driver.ParseCommand(_wizard.Target, quiet, preferSilent);
            if (cmd is null)
            {
                Status = "Aucune variante silencieuse — désinstallation interactive requise.";
                cmd = _driver.ParseCommand(_wizard.Target, quiet, preferSilent: false);
                if (cmd is null)
                {
                    Status = "Impossible de parser la commande de désinstallation.";
                    Outcome = new UninstallOutcome(UninstallStatus.ParseFailure, null, TimeSpan.Zero, 0, "ParseCommand returned null twice.");
                    _wizard.UninstallOutcome = Outcome;
                    return;
                }
            }
            Status = $"Lancement : {cmd.ExecutablePath} {cmd.Arguments}";
            var progress = new Progress<string>(line =>
            {
                OutputLines.Add(line);
                if (OutputLines.Count > 1000) OutputLines.RemoveAt(0);
            });
            Outcome = await _driver.RunAsync(cmd, progress, _cts.Token);
            _wizard.UninstallOutcome = Outcome;
            Status = $"Terminé : {Outcome.Status} (code={Outcome.ExitCode}) en {Outcome.Elapsed.TotalSeconds:F1}s";
        }
        catch (OperationCanceledException) { Status = "Annulé."; }
        catch (Exception ex) { Status = $"Erreur : {ex.Message}"; }
        finally { IsRunning = false; NextCommand.NotifyCanExecuteChanged(); }
    }

    private static string? ReadQuietUninstallString(InstalledProgram p) {
        try {
            using var hive = Microsoft.Win32.RegistryKey.OpenBaseKey(
                p.Hive is RegistryHive.LocalMachine64 or RegistryHive.LocalMachine32
                    ? Microsoft.Win32.RegistryHive.LocalMachine
                    : Microsoft.Win32.RegistryHive.CurrentUser,
                p.Hive is RegistryHive.LocalMachine64 or RegistryHive.CurrentUser64
                    ? Microsoft.Win32.RegistryView.Registry64
                    : Microsoft.Win32.RegistryView.Registry32);
            using var key = hive.OpenSubKey(@"SOFTWARE\Microsoft\Windows\CurrentVersion\Uninstall\" + p.RegistryKey, writable: false);
            return key?.GetValue("QuietUninstallString") as string;
        } catch { return null; }
    }
    ```

    ```csharp
    [RelayCommand]
    private async Task ScanAsync()
    {
        IsScanning = true;
        Findings.Clear();
        try
        {
            var target = new ResidueScanTarget(_wizard.Target.DisplayName, _wizard.Target.Publisher, _wizard.Target.InstallLocation, _wizard.Target.RegistryKey);
            var progress = new Progress<string>(area => CurrentArea = area);
            var scanned = await _residueScanner.ScanAsync(target, _wizard.Trace, progress, _cts.Token);
            foreach (var f in scanned) Findings.Add(f);

            // Merge publisher-rule-derived candidates
            foreach (var match in _wizard.MatchedRules)
            {
                foreach (var template in match.Rule.FilesystemPaths)
                {
                    var expanded = _ruleEngine.ExpandTokens(template, _wizard.Target.Publisher, _wizard.Target.DisplayName);
                    if (string.IsNullOrWhiteSpace(expanded)) continue;
                    if (!DiskScout.Helpers.ResiduePathSafety.IsSafeToPropose(expanded)) continue;
                    var exists = System.IO.Directory.Exists(expanded) || System.IO.File.Exists(expanded);
                    var trust = exists ? ResidueTrustLevel.HighConfidence : ResidueTrustLevel.LowConfidence;
                    var size = exists ? MeasureBytesIfDir(expanded) : 0;
                    Findings.Add(new ResidueFinding(ResidueCategory.Filesystem, expanded, size, $"Règle éditeur '{match.Rule.Id}'", trust, ResidueSource.PublisherRule));
                }
                foreach (var template in match.Rule.RegistryPaths)
                {
                    var expanded = _ruleEngine.ExpandTokens(template, _wizard.Target.Publisher, _wizard.Target.DisplayName);
                    if (!DiskScout.Helpers.ResiduePathSafety.IsSafeToPropose(expanded)) continue;
                    Findings.Add(new ResidueFinding(ResidueCategory.Registry, expanded, 0, $"Règle éditeur '{match.Rule.Id}'", ResidueTrustLevel.MediumConfidence, ResidueSource.PublisherRule));
                }
            }

            _wizard.AllResidueFindings.AddRange(Findings);
        }
        catch (OperationCanceledException) { }
        finally { IsScanning = false; NextCommand.NotifyCanExecuteChanged(); }
    }
    ```

    ```csharp
    public ConfirmDeleteStepViewModel(UninstallWizardViewModel wizard, IFileDeletionService deletion, Serilog.ILogger logger)
    {
        _wizard = wizard; _deletion = deletion; _logger = logger;
        BuildTree();
    }

    private void BuildTree()
    {
        Roots.Clear();
        foreach (var grp in _wizard.AllResidueFindings.GroupBy(f => f.Category))
        {
            var category = new ResidueTreeNode {
                Label = $"{CategoryLabel(grp.Key)} ({grp.Count()})",
                Category = grp.Key,
                IsChecked = false,
            };
            long catTotal = 0;
            foreach (var f in grp.OrderByDescending(x => x.SizeBytes))
            {
                if (!DiskScout.Helpers.ResiduePathSafety.IsSafeToPropose(f.Path)) continue; // defense-in-depth
                var leaf = new ResidueTreeNode {
                    Label = f.Path, Path = f.Path, SizeBytes = f.SizeBytes,
                    Category = f.Category, Trust = f.Trust, Source = f.Source, Reason = f.Reason,
                    IsChecked = false,
                };
                leaf.PropertyChanged += (_, e) => { if (e.PropertyName == nameof(ResidueTreeNode.IsChecked)) RecomputeTotals(); };
                category.Children.Add(leaf);
                catTotal += f.SizeBytes;
            }
            if (category.Children.Count > 0)
            {
                category.PropertyChanged += (_, e) => { if (e.PropertyName == nameof(ResidueTreeNode.IsChecked)) RecomputeTotals(); };
                Roots.Add(category);
            }
        }
    }

    private void RecomputeTotals()
    {
        long bytes = 0; int n = 0;
        foreach (var root in Roots) foreach (var leaf in root.EnumerateLeavesChecked()) { bytes += leaf.SizeBytes; n++; }
        TotalSelectedBytes = bytes; SelectedCount = n;
        ConfirmCommand.NotifyCanExecuteChanged();
    }

    [RelayCommand(CanExecute = nameof(CanConfirm))]
    private async Task ConfirmAsync()
    {
        var leaves = Roots.SelectMany(r => r.EnumerateLeavesChecked()).ToList();
        var paths = leaves.Where(l => DiskScout.Helpers.ResiduePathSafety.IsSafeToPropose(l.Path!)).Select(l => l.Path!).ToList();
        if (paths.Count == 0) return;

        var sizeStr = DiskScout.Helpers.ByteFormat.Fmt(TotalSelectedBytes);
        var msg = $"Supprimer DÉFINITIVEMENT {paths.Count} élément(s) ({sizeStr}) ?\n\nCette action est IRRÉVERSIBLE — pas de quarantaine, pas de corbeille.\n\nConfirmer ?";
        var res = System.Windows.MessageBox.Show(msg, "DiskScout — confirmer la suppression définitive",
            System.Windows.MessageBoxButton.YesNo, System.Windows.MessageBoxImage.Stop, System.Windows.MessageBoxResult.No);
        if (res != System.Windows.MessageBoxResult.Yes) return;

        try
        {
            DeletionOutcome = await _deletion.DeleteAsync(paths, DiskScout.Helpers.DeleteMode.Permanent, default);
            _wizard.DeletionOutcome = DeletionOutcome; // see Plan 06 for property addition
            _wizard.CurrentStep = WizardStep.Done;
            DiskScout.Helpers.DeletePrompt.ShowResult(DeletionOutcome, DiskScout.Helpers.DeleteMode.Permanent);
        }
        catch (Exception ex) { _logger.Error(ex, "Wizard deletion failed"); }
    }

    private bool CanConfirm() => SelectedCount > 0;
    ```
